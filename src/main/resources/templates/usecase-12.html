ATM & Branch Finder (Geospatial)

How it works

  1. Data ingestion: ATMs and branches loaded with coordinates via GEOADD (native) and JSON.SET (RQE approach).
  2. User selects location: Click the map or pick a preset location in Madrid.
  3. Radius search: GEOSEARCH finds nearby points by distance. FT.SEARCH with GEO filter adds type/service filtering.
  4. Results: Markers on map + sorted list with distance, type, and services.

# ═══ Approach 1: Native Geospatial ═══

# Add ATM locations — GEOADD key lng lat member
GEOADD workshop:geo:atms -3.7038 40.4168 "atm-001"
GEOADD workshop:geo:atms -3.7056 40.4200 "atm-002"
GEOADD workshop:geo:atms -3.6931 40.4194 "branch-001"

# Store metadata in separate Hash
HSET workshop:geo:meta:atm-001 name "Cajero Sol" \
  type atm address "Puerta del Sol 1" hours 24h \
  services "withdrawal,deposit,balance"

# Find ATMs within 2km of Puerta del Sol
GEOSEARCH workshop:geo:atms FROMLONLAT -3.7038 40.4168 \
  BYRADIUS 2 km ASC COUNT 10 WITHDIST
# → 1) "atm-001" "0.0000"
# → 2) "atm-009" "0.3150"
# → 3) "branch-007" "0.4820"

# Distance between two points
GEODIST workshop:geo:atms atm-001 branch-001 km
# → "0.9510"

# ═══ Approach 2: JSON + Redis Query Engine ═══

# Store rich documents with JSON.SET
JSON.SET workshop:geo:branch:atm-001 $ '{
  "name": "Cajero Sol", "type": "atm",
  "address": "Puerta del Sol 1, Madrid",
  "services": ["withdrawal","deposit","balance"],
  "hours": "24h",
  "location": "-3.7038,40.4168"
}'

# Create index with GEO + TAG fields
FT.CREATE idx:branches ON JSON
  PREFIX 1 workshop:geo:branch:
  SCHEMA
    $.location AS location GEO
    $.type AS type TAG
    $.services[*] AS services TAG
    $.name AS name TEXT
    $.hours AS hours TAG

# Geo search — all within 2km
FT.SEARCH idx:branches
  "@location:[-3.7038 40.4168 2 km]"

# Geo + type filter — only ATMs within 2km
FT.SEARCH idx:branches
  "@location:[-3.7038 40.4168 2 km] @type:{atm}"

# Geo + type + service filter
FT.SEARCH idx:branches
  "@location:[-3.7038 40.4168 2 km] @type:{branch} @services:{advisor}"
// ═══ Approach 1: Native Geospatial ═══

// GEOADD — add locations with coordinates
redis.opsForGeo().add("workshop:geo:atms",
    new Point(-3.7038, 40.4168), "atm-001");

// GEOSEARCH — find within radius, sorted by distance
GeoResults<GeoLocation<String>> results =
    redis.opsForGeo().search("workshop:geo:atms",
        GeoReference.fromCoordinate(-3.7038, 40.4168),
        new Distance(2, Metrics.KILOMETERS),
        GeoSearchCommandArgs.newGeoSearchArgs()
            .includeDistance()
            .sortAscending()
            .limit(20));

// Enrich with metadata from Hash
for (var r : results) {
    String id = r.getContent().getName();
    Map meta = redis.opsForHash()
        .entries("workshop:geo:meta:" + id);
    // meta has: name, type, address, services
}
