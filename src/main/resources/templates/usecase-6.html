1. Velocity Tracking (Sorted Set)

// ZADD — record each transaction with timestamp as score
String velocityKey = "workshop:fraud:velocity:" + cardNumber;
long now = Instant.now().toEpochMilli();

redis.opsForZSet().add(velocityKey, txId, now);

// ZRANGEBYSCORE — count txs in last 5 minutes
long windowStart = now - (300 * 1000); // 5 min ago
Set<String> recentTxs = redis.opsForZSet()
    .rangeByScore(velocityKey, windowStart, now);

int velocityCount = recentTxs.size();
// velocityCount > 5 → HIGH RISK!

// Clean expired entries
redis.opsForZSet()
    .removeRangeByScore(velocityKey, 0, windowStart - 1);

2. Redis CLI — Velocity Check

# Add transaction with timestamp as score
ZADD workshop:fraud:velocity:4000-1234 1713045600000 "tx-abc123"
# → (integer) 1

# Count transactions in last 5 minutes
ZRANGEBYSCORE workshop:fraud:velocity:4000-1234 1713045300000 1713045600000
# → 1) "tx-prev001"
# → 2) "tx-prev002"
# → 3) "tx-abc123"
# 3 transactions → elevated velocity

# Clean old entries
ZREMRANGEBYSCORE workshop:fraud:velocity:4000-1234 0 1713045299999

3. Event Stream (Redis Streams)

// XADD — log every fraud evaluation to a stream
Map<String, String> entry = Map.of(
    "txId", txId,
    "card", cardNumber,
    "amount", String.valueOf(amount),
    "riskScore", String.valueOf(riskScore),
    "riskLevel", riskLevel
);

redis.opsForStream().add(
    StreamRecords.string(entry)
        .withStreamKey("workshop:fraud:stream"));

// XREVRANGE — read most recent evaluations
redis.opsForStream().reverseRange(
    "workshop:fraud:stream",
    Range.unbounded(),
    Limit.limit().count(20));

4. Redis CLI — Stream Commands

# Log a fraud evaluation event
XADD workshop:fraud:stream * txId "tx-abc123" card "4000-1234" amount "15000.00" riskScore "75" riskLevel "CRITICAL"
# → "1713045600000-0"

# Read last 5 evaluations
XREVRANGE workshop:fraud:stream + - COUNT 5

# Stream length
XLEN workshop:fraud:stream

# Trim to last 50 entries
XTRIM workshop:fraud:stream MAXLEN ~ 50
