Document Search (Full-text + Vector)

How it works

  1. When a client sends a request to the app, a response within 100 milliseconds is guaranteed.
  2. The app requests data from Redis as the document database. Its built-in Redis Query Engine retrieves it in under 10 milliseconds.

// 1. Store regulation doc — JSON.SET
String key = "workshop:docs:regulation:REG001";
String json = """
  { "id": "REG001",
    "title": "PSD2 Strong Customer Auth",
    "category": "PSD2",
    "summary": "SCA requirements...",
    "content": "Full text...",
    "vector": [0.023, -0.118, ...] }
  """;

// JSON.SET — store as RedisJSON document
redis.execute(conn -> conn.execute(
    "JSON.SET", key, "$", json));

// 2. Create search index — FT.CREATE
// TEXT fields for full-text, TAG for filters,
// VECTOR HNSW for semantic similarity
redis.execute(conn -> conn.execute(
  "FT.CREATE", "idx:regulations",
  "ON", "JSON",
  "PREFIX", "1", "workshop:docs:regulation:",
  "SCHEMA",
    "$.title", "AS", "title", "TEXT",
      "WEIGHT", "2.0",
    "$.category", "AS", "category", "TAG",
    "$.summary", "AS", "summary", "TEXT",
    "$.content", "AS", "content", "TEXT",
    "$.vector", "AS", "vector", "VECTOR",
      "HNSW", "6",
      "TYPE", "FLOAT32",
      "DIM", "768",
      "DISTANCE_METRIC", "COSINE"));

// 3. Full-text search — FT.SEARCH
redis.execute(conn -> conn.execute(
  "FT.SEARCH", "idx:regulations",
  "payment services",
  "RETURN", "4",
    "title", "category", "summary", "tags",
  "LIMIT", "0", "10"));

// 4. Vector similarity (KNN) — FT.SEARCH
byte[] queryVec = embedQuery("data protection");
redis.execute(conn -> conn.execute(
  "FT.SEARCH", "idx:regulations",
  "*=>[KNN 10 @vector $BLOB AS score]",
  "RETURN", "5",
    "title", "category", "summary", "score",
  "SORTBY", "score",
  "PARAMS", "2", "BLOB", queryVec,
  "DIALECT", "2"));

// 5. Hybrid: text filter + KNN
redis.execute(conn -> conn.execute(
  "FT.SEARCH", "idx:regulations",
  "(payment)=>[KNN 5 @vector $BLOB AS score]",
  "RETURN", "5",
    "title", "category", "summary", "score",
  "SORTBY", "score",
  "PARAMS", "2", "BLOB", queryVec,
  "DIALECT", "2"));

                    


                    

                        
# 1. Store document as JSON
JSON.SET workshop:docs:regulation:REG001 $ '{
  "id": "REG001",
  "title": "PSD2 Strong Customer Auth",
  "category": "PSD2",
  "summary": "SCA requirements for payments",
  "content": "Full regulation text...",
  "vector": [0.023, -0.118, ...]
}'

# 2. Create index: TEXT + TAG + VECTOR on JSON
FT.CREATE idx:regulations ON JSON \
  PREFIX 1 workshop:docs:regulation: \
  SCHEMA \
    $.title AS title TEXT WEIGHT 2.0 \
    $.category AS category TAG \
    $.summary AS summary TEXT \
    $.content AS content TEXT \
    $.vector AS vector VECTOR HNSW 6 \
      TYPE FLOAT32 DIM 768 \
      DISTANCE_METRIC COSINE

# 3. Full-text search
FT.SEARCH idx:regulations "payment services" \
  RETURN 4 title category summary tags \
  LIMIT 0 10

# 4. Vector similarity (KNN)
# $BLOB = binary float32 query vector
FT.SEARCH idx:regulations \
  "*=>[KNN 10 @vector $BLOB AS score]" \
  RETURN 5 title category summary tags score \
  SORTBY score \
  PARAMS 2 BLOB $query_vector \
  DIALECT 2

# 5. Hybrid: text pre-filter + KNN
FT.SEARCH idx:regulations \
  "(payment)=>[KNN 5 @vector $BLOB AS score]" \
  RETURN 5 title category summary tags score \
  SORTBY score \
  PARAMS 2 BLOB $query_vector \
  DIALECT 2

# 6. Filter by category TAG
FT.SEARCH idx:regulations \
  "@category:{PSD2}" \
  RETURN 3 title summary tags \
  LIMIT 0 10
