Document Search (Full-text + Vector)

How it works
  1. Create & Store — Documents are stored as native JSON in Redis using JSON.SET, with full JSONPath support for partial reads and updates.
  2. Query & Filter — The Redis Query Engine indexes JSON fields (TEXT, TAG, VECTOR) for instant queries using FT.SEARCH with field filters.
  3. Search — Full-text, vector similarity (KNN), and hybrid search all run in under 10ms on the same indexed data.

// === CRUD Operations ===

// 1. CREATE — JSON.SET (store document)
String key = "uc8:doc:psd2:chunk:0";
String json = """
  { "id": "psd2:chunk:0",
    "title": "PSD2 - Payment Services Directive 2",
    "category": "PSD2",
    "summary": "SCA requirements...",
    "content": "Full text...",
    "vector": [0.023, -0.118, ...] }
  """;
redis.execute(conn -> conn.execute(
    "JSON.SET", key, "$", json));

// 2. READ full doc — JSON.GET
redis.execute(conn -> conn.execute(
    "JSON.GET", key, "$"));

// 3. READ specific field — JSON.GET + path
redis.execute(conn -> conn.execute(
    "JSON.GET", key, "$.title"));

// 4. READ multiple docs — JSON.MGET
redis.execute(conn -> conn.execute(
    "JSON.MGET", key1, key2, key3, "$"));

// 5. QUERY by field — FT.SEARCH + filter
redis.execute(conn -> conn.execute(
  "FT.SEARCH", "idx:uc8:documents",
  "@category:{PSD2}",
  "RETURN", "4",
    "title", "category", "summary", "tags",
  "LIMIT", "0", "10"));

// === Search Operations ===

// 6. Full-text search — FT.SEARCH
redis.execute(conn -> conn.execute(
  "FT.SEARCH", "idx:uc8:documents",
  "payment services",
  "RETURN", "4",
    "title", "category", "summary", "tags",
  "LIMIT", "0", "10"));

// 7. Vector similarity (KNN)
byte[] queryVec = embedQuery("data protection");
redis.execute(conn -> conn.execute(
  "FT.SEARCH", "idx:uc8:documents",
  "*=>[KNN 10 @vector $BLOB AS score]",
  "RETURN", "5",
    "title", "category", "summary", "score",
  "SORTBY", "score",
  "PARAMS", "2", "BLOB", queryVec,
  "DIALECT", "2"));

// 8. Hybrid: text filter + KNN
redis.execute(conn -> conn.execute(
  "FT.SEARCH", "idx:uc8:documents",
  "(payment)=>[KNN 5 @vector $BLOB AS score]",
  "RETURN", "5",
    "title", "category", "summary", "score",
  "SORTBY", "score",
  "PARAMS", "2", "BLOB", queryVec,
  "DIALECT", "2"));

                    


                    

                        
# === CRUD Operations ===

# 1. CREATE — store document as JSON
JSON.SET uc8:doc:psd2:chunk:0 $ '{
  "id": "psd2:chunk:0",
  "title": "PSD2 - Payment Services Directive 2",
  "category": "PSD2",
  "summary": "SCA requirements...",
  "content": "Full regulation text..."
}'

# 2. READ full document
JSON.GET uc8:doc:psd2:chunk:0 $

# 3. READ specific field (JSONPath)
JSON.GET uc8:doc:psd2:chunk:0 $.title
JSON.GET uc8:doc:psd2:chunk:0 $.summary

# 4. READ multiple documents at once
JSON.MGET uc8:doc:psd2:chunk:0 \
  uc8:doc:gdpr:chunk:0 $

# 5. QUERY by field — TAG filter
FT.SEARCH idx:uc8:documents "@category:{PSD2}" \
  RETURN 4 title category summary tags \
  LIMIT 0 10

# === Search Operations ===

# 6. Full-text search
FT.SEARCH idx:uc8:documents "payment services" \
  RETURN 4 title category summary tags \
  LIMIT 0 10

# 7. Vector similarity (KNN)
FT.SEARCH idx:uc8:documents \
  "*=>[KNN 10 @vector $BLOB AS score]" \
  RETURN 5 title category summary tags score \
  SORTBY score \
  PARAMS 2 BLOB $query_vector \
  DIALECT 2

# 8. Hybrid: text pre-filter + KNN
