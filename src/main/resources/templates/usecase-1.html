Code Showcase

// 1. Create session — HSET + EXPIRE
String sessionKey = "workshop:session:user:" + username;

// Store session data as a Redis Hash
Map<String, String> session = Map.of(
    "username",  username,
    "fullName",  "Ana García López",
    "role",      "Premium Client",
    "accountId", "ES76-0128-0001-23-0100012345",
    "token",     UUID.randomUUID().toString(),
    "createdAt", String.valueOf(System.currentTimeMillis())
);

// HSET — write all fields atomically
redis.opsForHash().putAll(sessionKey, session);

// EXPIRE — auto-delete after 300 seconds
redis.expire(sessionKey, 300, TimeUnit.SECONDS);

// 2. Store auth token — SET with EX
String tokenKey = "workshop:session:token:" + token;
redis.opsForValue()
     .set(tokenKey, username, 300, TimeUnit.SECONDS);

// 3. Read session — HGETALL
Map<Object, Object> data =
    redis.opsForHash().entries(sessionKey);

// 4. Check remaining TTL — TTL command
Long ttl = redis.getExpire(sessionKey,
                           TimeUnit.SECONDS);

// 5. Validate token — GET
String user = redis.opsForValue().get(tokenKey);

// 6. Logout — DEL session + token
redis.delete(sessionKey);
redis.delete(tokenKey);
# 1. Create session Hash
HSET workshop:session:user:user1 \
  username "user1" \
  fullName "Ana García López" \
  role "Premium Client" \
  accountId "ES76-0128-0001-23-0100012345" \
  token "a1b2c3d4-e5f6-7890-abcd-ef1234567890" \
  createdAt "1713045600000"

# 2. Set TTL (300 seconds = 5 min)
EXPIRE workshop:session:user:user1 300

# 3. Store auth token → username mapping
SET workshop:session:token:a1b2c3d4-... \
  "user1" EX 300

# 4. Read full session
HGETALL workshop:session:user:user1

# 5. Read single field
HGET workshop:session:user:user1 role

# 6. Check remaining TTL
TTL workshop:session:user:user1

# 7. Validate token
GET workshop:session:token:a1b2c3d4-...

# 8. Logout — delete both keys
DEL workshop:session:user:user1
DEL workshop:session:token:a1b2c3d4-...
