UC16

AI Gateway

How it works
  1. Semantic routing stores model capability vectors in Redis and uses KNN search to choose the best provider for each prompt.
  2. Semantic cache checks for a similar request per model before any mock LLM work happens.
  3. Per-model rate limiting uses INCR + EXPIRE so each provider gets its own request budget.
  4. Cost tracking accumulates estimated token usage and spend in Redis Hashes per session.
  5. Observability appends every request to a Redis Stream so the UI can show recent gateway traffic.

// 1. Semantic routing — model capability vectors
RouteDecision route = routeQuery(query, queryVector);

// 2. Semantic cache — only within the selected model tier
CacheResult cache = checkSemanticCache(route.model(), queryVector);

// 3. Per-model rate limit — INCR + EXPIRE window
Map<String, Object> limit = consumeRateLimit(route.model());

// 4. Session usage + cost tracking — Hash increments
redis.opsForHash().increment("uc16:usage:session:" + sessionId,
    route.model().tag() + ":costUsd", estimatedCostUsd);

// 5. Observability — append one event per request
redis.opsForStream().add(StreamRecords.string(Map.of(
    "model", route.model().label(),
    "cacheHit", String.valueOf(cache.hit()),
    "latencyMs", String.valueOf(totalMs),
    "costUsd", String.format(Locale.US, "%.6f", estimatedCostUsd)
)).withStreamKey("uc16:stream:gateway"));
# 1. Route index — capability vectors per model
FT.CREATE idx:uc16:routes \
  ON HASH PREFIX 1 uc16:route:model: \
  SCHEMA \
    modelId TEXT modelTag TAG label TEXT capability TEXT rationale TEXT \
    vector VECTOR HNSW 6 TYPE FLOAT32 DIM 10 DISTANCE_METRIC COSINE

# 2. Semantic route selection (KNN)
FT.SEARCH idx:uc16:routes \
  "*=>[KNN 1 @vector $BLOB]" \
  PARAMS 2 BLOB "<query_vector>" \
  RETURN 5 modelId modelTag label capability rationale \
  DIALECT 2

# 3. Gateway-level semantic cache per model
FT.SEARCH idx:uc16:cache \
  "@modelTag:{gpt4o}=>[KNN 1 @vector $BLOB]" \
  PARAMS 2 BLOB "<query_vector>" \
  RETURN 4 question response modelId createdAt \
  DIALECT 2

# 4. Per-model fixed window rate limit
INCR uc16:ratelimit:gpt4o
EXPIRE uc16:ratelimit:gpt4o 60

# 5. Session usage + cost
HINCRBYFLOAT uc16:usage:session:gw-001 gpt4o:costUsd 0.0043
HINCRBY uc16:usage:session:gw-001 totalTokens 842

# 6. Observability stream
XADD uc16:stream:gateway * \
  model "GPT-4o" cacheHit "false" rateLimited "false" \
  latencyMs "214" costUsd "0.0043" query "Explain Basel III..."

# 7. Inspect recent gateway traffic
XREVRANGE uc16:stream:gateway + - COUNT 10
