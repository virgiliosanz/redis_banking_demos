// 1. Aggregate from multiple DBs — HSET
String key = "workshop:profile:user:" + userId;

Map<String, String> profile = new HashMap<>();

// From Accounts DB
profile.put("account_number", "ES76-0128...");
profile.put("account_balance", "12450.75");
profile.put("account_type", "Premium Savings");

// From Activity DB
profile.put("activity_lastLogin", "2024-04-12");
profile.put("activity_recentTxCount", "14");

// From Preferences DB
profile.put("pref_language", "es");
profile.put("pref_notifications", "email");

// HSET — store all fields in one call
redis.opsForHash().putAll(key, profile);

// EXPIRE — auto-cleanup
redis.expire(key, 600, TimeUnit.SECONDS);

// 2. Read full profile — HGETALL
Map<Object, Object> data =
    redis.opsForHash().entries(key);

// 3. Update single field — HSET
redis.opsForHash()
    .put(key, "pref_language", "en");

// 4. Atomic counter — HINCRBY
redis.opsForHash()
    .increment(key, "activity_loginCount", 1);
