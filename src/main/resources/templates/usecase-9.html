AI Banking Assistant (Memory + RAG)

How it works

Agent Memory
  1. User makes a request to an agent.
  2. The agent generates a plan.
  3. The agent calls tools or an LLM for information.
  4. The agent stores information in short-term memory.
  5. Some information will be stored for long-term, while other info will only be saved temporarily.
Retrieval-Augmented Generation (RAG)
  1. A user asks a question.
  2. The app retrieves the relevant document or chunk from Redis.
  3. The question and relevant document are sent to an LLM.
  4. The LLM generates an answer using its knowledge plus the retrieved information.
  5. The app returns the answer to the user.

// 1. Short-term Memory — HSET + EXPIRE
// Store conversation context as a Hash
String convKey = "workshop:assistant:conversation:"
    + sessionId;

Map<String, String> convData = Map.of(
    "messages", objectMapper.writeValueAsString(
        messageHistory),
    "created_at", Instant.now().toString(),
    "last_active", Instant.now().toString(),
    "user_name", userName
);
redis.opsForHash().putAll(convKey, convData);
// TTL: auto-expire after 10 min of inactivity
redis.expire(convKey, 600, TimeUnit.SECONDS);

// 2. Long-term Memory — Hash + Vector
// Store past interaction with embedding vector
String memKey = "workshop:assistant:memory:mem-001";
redis.opsForHash().putAll(memKey, Map.of(
    "summary", "Asked about wire transfer fees",
    "detail", "Customer inquired about SWIFT...",
    "date", "2024-03-15",
    "tags", "transfer,international,fees"
));
// Store vector as raw FLOAT32 bytes
conn.hashCommands().hSet(
    memKey.getBytes(),
    "vector".getBytes(),
    vectorToBytes(embedding)  // 768-dim float[]
);

# 1. Short-term Memory — conversation context
HSET workshop:assistant:conversation:sess-001 \
  messages '[{"role":"user","content":"..."}]' \
  created_at "2024-03-15T10:00:00Z" \
  last_active "2024-03-15T10:05:00Z" \
  user_name "Demo User"

# Auto-expire after 10 minutes
EXPIRE workshop:assistant:conversation:sess-001 600

# Read conversation
HGETALL workshop:assistant:conversation:sess-001

# Check remaining TTL
TTL workshop:assistant:conversation:sess-001

# 2. Long-term Memory — store with vector
HSET workshop:assistant:memory:mem-001 \
  summary "Asked about wire transfer fees" \
  detail "Customer inquired about SWIFT..." \
  date "2024-03-15" \
  tags "transfer,international,fees" \
  vector "<768-dim FLOAT32 binary blob>"

# 3. Create vector index for memories
FT.CREATE idx:assistant_memory \
  ON HASH PREFIX 1 workshop:assistant:memory: \
  SCHEMA \
    summary TEXT \
    tags TAG SEPARATOR "," \
    date TEXT \
    vector VECTOR HNSW 6 \
      TYPE FLOAT32 DIM 768 \
      DISTANCE_METRIC COSINE

# 4. KNN search — find similar memories
FT.SEARCH idx:assistant_memory \
  "*=>[KNN 3 @vector $BLOB]" \
  PARAMS 2 BLOB "<query_vector>" \
  RETURN 3 summary detail date \
  DIALECT 2

# 5. RAG knowledge base index
FT.CREATE idx:assistant_kb \
  ON HASH PREFIX 1 workshop:assistant:kb: \
  SCHEMA \
    title TEXT \
    content TEXT \
    tags TAG SEPARATOR "," \
    vector VECTOR HNSW 6 \
      TYPE FLOAT32 DIM 768 \
      DISTANCE_METRIC COSINE

# 6. KNN search — retrieve KB articles
FT.SEARCH idx:assistant_kb \
  "*=>[KNN 3 @vector $BLOB]" \
  PARAMS 2 BLOB "<query_vector>" \
  RETURN 3 title content tags \
  DIALECT 2

# Memory patterns summary:
# SHORT-TERM: Hash + TTL (10 min window)
# LONG-TERM:  Hash + Vector (KNN search)
# RAG:        Hash + Vector (KNN search)
# All served from Redis in < 5ms total
