// ═══ Fixed-Window Rate Limiting ═══
// Pattern: INCR + EXPIRE
// Simple, efficient, atomic counter per time window

String key = "workshop:ratelimit:" + clientId;

// INCR atomically increments (creates key=1 if absent)
Long count = redis.opsForValue().increment(key);

// Set TTL only on the FIRST request of a new window
if (count == 1) {
    redis.expire(key, 60, TimeUnit.SECONDS);
}

// Check if client exceeded the limit
boolean allowed = count <= MAX_REQUESTS;  // e.g. 10
long remaining = Math.max(0, MAX_REQUESTS - count);

if (!allowed) {
    // Return 429 Too Many Requests
    // Include Retry-After header (TTL of the key)
    Long ttl = redis.getExpire(key, TimeUnit.SECONDS);
    response.setHeader("Retry-After", ttl.toString());
}
// ═══ Sliding-Window Rate Limiting ═══
// Pattern: Sorted Set (ZADD + ZRANGEBYSCORE + ZCARD)
// More precise than fixed-window, no boundary burst

String key = "workshop:ratelimit:sliding:" + clientId;
long now = System.currentTimeMillis();
long windowStart = now - (WINDOW_SECONDS * 1000);

// 1. Remove expired entries outside the window
redis.opsForZSet().removeRangeByScore(key, 0, windowStart);

// 2. Count current requests in the window
Long count = redis.opsForZSet().zCard(key);

if (count != null && count < MAX_REQUESTS) {
    // 3. Add current request (score = timestamp)
    redis.opsForZSet().add(key, UUID.randomUUID().toString(), now);

    // 4. Set TTL as safety net for cleanup
    redis.expire(key, WINDOW_SECONDS + 1, TimeUnit.SECONDS);

    return true;  // Allowed
} else {
    return false; // Rate limit exceeded — 429
}

// Advantage: no "boundary burst" problem
// A client can't get 2x the limit at window boundaries
# ═══ Fixed-Window with Redis CLI ═══

# Increment counter (creates if not exists)
INCR workshop:ratelimit:fintech-app-001
# → (integer) 1

# Set expiry on first request
EXPIRE workshop:ratelimit:fintech-app-001 60
# → (integer) 1

# Check current count
GET workshop:ratelimit:fintech-app-001
# → "5"

# Check TTL (time until window resets)
TTL workshop:ratelimit:fintech-app-001
# → (integer) 42

# ═══ Sliding-Window with Redis CLI ═══

# Add request with timestamp as score
ZADD workshop:ratelimit:sliding:fintech-app-001 1713045600000 "req-uuid-1"

# Remove expired entries
ZREMRANGEBYSCORE workshop:ratelimit:sliding:fintech-app-001 0 1713045540000

# Count requests in current window
ZCARD workshop:ratelimit:sliding:fintech-app-001
# → (integer) 3
