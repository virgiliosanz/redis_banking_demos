// 1. Store client features — HSET (bulk)
String key = "workshop:features:client:C1001";

Map<String, String> features = Map.of(
    "tx_count_1h",         "2",
    "tx_count_24h",        "5",
    "tx_amount_avg_24h",   "120.50",
    "tx_amount_max_24h",   "250.00",
    "distinct_countries_7d","1",
    "distinct_devices_30d", "2",
    "risk_score",           "12"
);

// HSET — write all features atomically
redis.opsForHash().putAll(key, features);

// 2. Read full feature vector — HGETALL
Map<Object, Object> allFeatures =
    redis.opsForHash().entries(key);
// Returns ALL fields in ~0.1ms (vs 50-200ms
// from a data warehouse query)

// 3. Atomic counter update — HINCRBY
// New transaction arrives → increment counters
redis.opsForHash().increment(key,
    "tx_count_1h", 1);
redis.opsForHash().increment(key,
    "tx_count_24h", 1);

// 4. Update computed features — HSET
redis.opsForHash().put(key,
    "tx_amount_avg_24h", "135.25");
redis.opsForHash().put(key,
    "risk_score", "15");

// 5. Pipeline for bulk updates
// (multiple HSET in one round-trip)
redis.executePipelined((RedisCallback<Object>) conn -> {
    byte[] k = key.getBytes();
    conn.hashCommands().hSet(k,
        "tx_count_1h".getBytes(), "3".getBytes());
    conn.hashCommands().hSet(k,
        "last_tx_timestamp".getBytes(),
        String.valueOf(now).getBytes());
    conn.hashCommands().hSet(k,
        "risk_score".getBytes(), "18".getBytes());
    return null;
});

// 6. Read single feature — HGET
String riskScore = (String) redis.opsForHash()
    .get(key, "risk_score");

                    


                    

                        
# 1. Store feature vector (bulk HSET)
HSET workshop:features:client:C1001 \
  tx_count_1h 2 \
  tx_count_24h 5 \
  tx_amount_avg_24h "120.50" \
  tx_amount_max_24h "250.00" \
  distinct_countries_7d 1 \
  distinct_devices_30d 2 \
  risk_score 12

# 2. Read full feature vector (~0.1ms)
HGETALL workshop:features:client:C1001

# 3. Atomic counter increment
HINCRBY workshop:features:client:C1001 \
  tx_count_1h 1
HINCRBY workshop:features:client:C1001 \
  tx_count_24h 1

# 4. Update computed features
HSET workshop:features:client:C1001 \
  tx_amount_avg_24h "135.25" \
  risk_score 15

# 5. Read single feature
HGET workshop:features:client:C1001 risk_score

# 6. Feature store vs data warehouse
# Redis Hash: HGETALL → ~0.1ms latency
# SQL query:  SELECT * FROM features
#             WHERE client_id = 'C1001'
#             → 50-200ms latency
#
# For real-time ML scoring, sub-ms
# feature retrieval is critical.
